
public class Solution : GuessGame {
    public int GuessNumber(int n) {
        int l=1,r=n;
        while(l<=r){
            int mid=l+(r-l)/2;
            int guessResult=guess(mid);
            if(guessResult==0)
                return mid;
            else if(guessResult==1)
                l=mid+1;
            else
                r=mid-1;
        }
        return n;
    }
}
