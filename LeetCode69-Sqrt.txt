public class Solution {
    public int MySqrt(int x) {
        if(x==1)
            return 1;
        long tempX=(long)x;
        int l=0,r=x/2;
        while(l<=r){
            int mid=l+(r-l)/2;
            long sqr=(long)mid*mid;
            if(sqr==tempX)
                return mid;
            else if(sqr>tempX)
                r=mid-1;
            else
                l=mid+1;
        }
        return r;
    }
}
