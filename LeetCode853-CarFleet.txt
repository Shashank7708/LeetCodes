public class Solution {
    public int CarFleet(int target, int[] position, int[] speed) {
        Dictionary<int,int> dict=new Dictionary<int,int>();
        for(int i=0;i<position.Length;i++){
            dict[position[i]]=speed[i];
        }
        int[] sortedPosition=new int[position.Length];
        Array.Copy(position, sortedPosition,position.Length);
        Array.Sort(sortedPosition);
        Array.Reverse(sortedPosition);
        double[] TimeToReach=new double[position.Length];

        for(int i=0;i<sortedPosition.Length;i++){
            var carPosition=sortedPosition[i];
            var carspeed=dict[carPosition];
            TimeToReach[i]=((double)target- (double)carPosition)/(double)carspeed;
        }

        double lastTime = 0;
        int fleet = 0;
        for(int i=0;i<TimeToReach.Length;i++){
            if(lastTime<TimeToReach[i]){
                fleet++;
                lastTime=TimeToReach[i];
            }
        }
        
        return fleet;
        
    }
}
