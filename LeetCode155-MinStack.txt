public class MinStack {
    private class Node{
        public int value;
        public int minValue;
        public Node next;
        public Node(int v, int minV,Node next){
            this.value=v;
            this.minValue=minV;
            this.next=next;
        }
    }

    Node head;
    int count=0;
    int minValue=int.MaxValue;
    public MinStack() {
        
    }
    public void Push(int val) {
        if(count==0){
            minValue=val;
            head=new Node(val,minValue,null);
        }
        else{
            minValue=Math.Min(minValue,val);
            var node=new Node(val,minValue,head);
            head=node;
        }
        count++;
    }
    public void Pop() {
        if(count==0)
            throw new Exception("Stack is Empty");
        head=head.next;
        count--;
       if(count==0){
        minValue=0;
        return;
       }
       else
        minValue=head.minValue;

    }
    public int Top() {
        if(head==null)
            throw new Exception("Stack is Empty");
        return head.value;
    }
    public int GetMin() {
        return head.minValue;
    }
